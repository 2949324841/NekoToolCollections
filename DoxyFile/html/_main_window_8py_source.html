111def search_and_scroll(table_view, column_name: str, target_value: str):
112 """根据列名和值搜索并跳转"""
113 model = table_view.model()

122 # 跳转并选中整行
123 table_view.scrollTo(index, QTableView.PositionAtCenter)

125 return True
